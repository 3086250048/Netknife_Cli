init_exp    => protocol_exp connect_exp
	        => protocol_exp connect_exp param_exp

connect_exp => address_exp
		    => address_exp port_exp

protocol_exp=> ssh | telnet | ftp | tftp | scp 

address_exp => ipv4_exp | ipv6_exp | domain_exp

ipv4_exp    => v4mini_reg address_split_exp v4mini_reg address_split_exp  v4mini_reg address_split_exp v4mini_reg

ipv6_exp    => address_split_exp v6mini_reg
	        => address_split_exp v6mini_reg ipv6_exp
	        => v6mini_reg address_split_exp v6mini_reg
	        => v6mini_reg address_split_exp v6mini_reg ipv6_exp

domain_exp  => address_split_exp sn_reg
		    => sn_reg address_split_exp sn_reg
		    => address_split_exp sn_exp domian_exp
		    => sn_reg address_split_exp sn_reg domian_exp
address_split_exp => . | : | :: 

port_exp => port_reg

param_exp   => inner_reg condition_reg sn_exp
	        => inner_reg condition_reg connect_exp
	        => inner_reg condition_reg sn_exp param_exp
	        => inner_reg condition_reg connect_exp param_exp
